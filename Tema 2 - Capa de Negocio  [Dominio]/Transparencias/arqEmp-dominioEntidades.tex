\begin{frame}[c]
    \frametitle{Objetivos del Tema}
    \begin{enumerate}[<+->]
         \item Comprender en profundidad cuáles son las responsabilidades de la capa de dominio.
         \item Conocer los principales patrones relacionados con el desarrollo de una capa de dominio.
         \item Conocer las ventajas e inconvenientes del patrón \emph{Domain Model}.
         \item Ser capaz de aplicar el patrón \emph{Domain Model}.
         \item Ser capaz de aplicar los principios de \emph{Domain-Driven Design}.
    \end{enumerate}
\end{frame}
